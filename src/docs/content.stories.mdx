## How to configure

Define classes in **src/assets/pages.styles.scss**.

<Source language='scss' code={dedent`
  /* Section content */
  .section-content {
    @apply py-20;              /* padding-top/bottom: 80px; */
    @apply px-6;               /* padding-left/right: 24px; */
    @media (min-width: theme('screens.md')) {
      @apply px-8;             /* padding-left/right: 32px; */
    }
    @media (min-width: theme('screens.lg')) {
      @apply py-24;            /* padding-top/bottom: 96px; */
    }
  }
  /* Section divider */
  .section-divider {
    @apply border-b;           /* border-bottom-width: 1px; */
    @apply border-primary-200; /* border-color: theme('colors.primary.200'); */
  }
  /* Content grid */
  .content-grid {
    @apply grid;               /* display: grid; */
    @apply grid-cols-4;        /* grid-template-columns: repeat(4, minmax(0, 1fr)); */
    @apply gap-x-6;            /* column-gap: 24px; */
    @apply gap-y-8;            /* row-gap: 32px; */
    @media (min-width: theme('screens.md')) {
      @apply grid-cols-12;     /* grid-template-columns: repeat(12, minmax(0, 1fr)); */
      @apply gap-x-8;          /* column-gap: 32px; */
    }
    @media (min-width: theme('screens.lg')) {
      @apply grid-cols-16;     /* grid-template-columns: repeat(16, minmax(0, 1fr)); */
      @apply gap-x-12;         /* column-gap: 48px; */
    }
    @media (min-width: 1441px) {
      max-width: 1440px;
      margin: 0 auto;
    }
  }
  /* Sub-section: spacing */
  section + section {

      @apply pt-20;            /* padding-top: 80px; */
    }
  }
`} />
